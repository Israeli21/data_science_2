# Part 3: Defining the Perceptron Model
# Complete the code of the Perceptron model based on what we've provided in the next slide
class Perceptron():
    def __init__(self, num_features, learning_rate = 0.1):
        self.num_features = num_features
        self.lr = learning_rate
        self.weights = np.zeros(num_features)  # Initialize weights to 0
        self.bias = 0.0                        # Initilize bias to 0

    def forward(self, x):
        linear = np.dot(x, self.weights) + self.bias
        predictions = np.where(linear >= 0, 1, 0)
        return predictions

    def backward(self, x, y):
        predictions = self.forward(x)
        error = y - predictions

        # Update weights and bias using perceptron rule
        for i in range(x.shape[0]):
            self.weights += self.lr * errors[i] * x[i]
            self.bias += self.lr * errors[i]
        return errors

    def train(self, x, y, epochs):
        for e in range(epochs):
            for i in range(y.shape[0]):
                prediction = self.forward(x[i])
                error = y[i] - prediction

                # weight and bias update (perceptron rule)
                self.weights += self.lr * error * x[i]
                self.bias += self.lr * error

    def evaluate(self, x, y):
        predictions = self.forward(x)
        accuracy = np.mean(predictions == y)
        return accuracy

# Part 6: Decision Boundary
# Plot the 2d Decision Boundary for your train set
# Plot the 2d Decision Boundary for your test set
def plot_decision_boundary(perceptron, X, y, title):
    # Scatter plot of points
    plt.figure(figsize=(8, 6))
    plt.scatter(X[:, 0], X[:, 1], c=y, cmap='bwr', alpha=0.7, edgecolors='k')

    # Get weights and bias
    w1, w2 = perceptron.weights
    b = perceptron.bias

    # Generate x values (span the feature space)
    x_min, x_max = X[:,0].min() - 1, X[:,0].max() + 1
    x_vals = np.linspace(x_min, x_max, 100)

    # Compute decision boundary line
    if w2 != 0:  # avoid division by zero
        y_vals = -(w1/w2) * x_vals - (b/w2)
        plt.plot(x_vals, y_vals, 'k--', linewidth=2, label="Decision Boundary")

    plt.title(title)
    plt.xlabel("Feature 1 (scaled)")
    plt.ylabel("Feature 2 (scaled)")
    plt.legend()
    plt.show()

# Plot for training set
plot_decision_boundary(perceptron, X_train_scaled, y_train, "Decision Boundary - Training Set")

# Plot for test set
plot_decision_boundary(perceptron, X_test_scaled, y_test, "Decision Boundary - Test Set")
